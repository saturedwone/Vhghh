🎮 PRIMER JUEGO
crearSprite("player", 100,300, 40,40, "#f97316");
setGravedad(0.8);
function onUpdate() {
  if(teclaPresionada("ArrowRight")) moverSprite("player",4,0);
  if(teclaPresionada("ArrowLeft")) moverSprite("player",-4,0);
  texto("Hola mundo", 20, 50);
}
🖼️ USAR IMÁGENES SUBIDAS

cargarImagenSprite("miSprite", "nombre_de_la_imagen.png");
